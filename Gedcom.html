def read(io)
  transmission = Transmission.new
  gedcom_parser = GedcomParser.new(transmission)
  
  io.each_line("\n") do |line|
    begin
      gedcom_parser.parse( io.lineno, line )
    rescue => exception
      puts "#{io.lineno}: #{exception} - " + line
      # raise exception
    end
  end
  
  @transmissions << transmission
end
